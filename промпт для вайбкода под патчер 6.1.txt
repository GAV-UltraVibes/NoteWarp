// [class-app>
class App {
  // [method-constructor>
  constructor() {
    this.data = [];
    this.init();
  }
  // <method-constructor]

  // [method-init>
  init() {
    this.loadData();
    this.render();
  }
  // <method-init]
}
// <class-app]

// [init-domready>
document.addEventListener('DOMContentLoaded', () => {
  window.app = new App();
});
// <init-domready]
